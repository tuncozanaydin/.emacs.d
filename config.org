(defun toa/tangle-config ()
  "Tangle config.org to config.el when saving config.org."
  (when (and buffer-file-name
             (string-equal (buffer-file-name) toa/config-org))
    (message "Tangling config.org...")
    (org-babel-tangle)))

(add-hook 'after-save-hook #'toa/tangle-config)

* Splash Screen
#+begin_src emacs-lisp :tangle config.el
(defvar toa/splash-text "Ξ  M  Λ  C  S")
(defvar toa/splash-svg-path "~/.emacs.d/blackhole-lines.svg")

(defun toa/splash-buffer ()
  "Get or create the splash buffer."
  (get-buffer-create "*splash*"))

(defun toa/splash-insert-centered (text)
  "Insert TEXT centered in the current buffer."
  (let* ((width (window-body-width))
         (padding (max 0 (/ (- width (length text)) 2))))
    (insert (make-string padding ?\s) text "\n")))

(defun toa/splash-insert-svg ()
  "Insert SVG image centered in the buffer."
  (when (and (display-graphic-p)
             (image-type-available-p 'svg)
             (file-readable-p (expand-file-name toa/splash-svg-path)))
    (let* ((svg-path (expand-file-name toa/splash-svg-path))
           (img-width 600) ;; Fixed width in pixels
           (img (create-image svg-path 'svg nil :width img-width))
           (img-size (image-size img t))
           (img-cols (max 1 (floor (/ (car img-size) (float (frame-char-width))))))
           (padding (max 0 (/ (- (window-body-width) img-cols) 2))))
      (insert (make-string padding ?\s))
      (insert-image img)
      (insert "\n\n"))))

(defun toa/splash-render ()
  "Render the splash screen."
  (with-current-buffer (toa/splash-buffer)
    (let ((inhibit-read-only t))
      (erase-buffer)
      (setq-local truncate-lines t
                  cursor-type nil
                  mode-line-format nil
                  header-line-format nil)
      (toa/splash-insert-svg)
      (toa/splash-insert-centered toa/splash-text)
      (insert "\n")
      (toa/splash-insert-centered
       (format "Loaded %d packages in %.2fs"
               (length package-activated-list)
               (float-time (time-subtract after-init-time before-init-time))))
      (goto-char (point-min))
      (read-only-mode 1))))

(defun toa/splash-on-resize (_frame)
  "Re-render splash screen when window size changes."
  (when (buffer-live-p (toa/splash-buffer))
    (dolist (win (get-buffer-window-list (toa/splash-buffer) nil t))
      (with-selected-window win
        (toa/splash-render)))))

(defun toa/splash-dismiss ()
  "Close the splash screen and remove resize hook."
  (interactive)
  (remove-hook 'window-size-change-functions #'toa/splash-on-resize)
  (when (buffer-live-p (toa/splash-buffer))
    (kill-buffer (toa/splash-buffer))))

(defun toa/splash-show ()
  "Show the splash screen."
  (interactive)
  (when (display-graphic-p)
    (switch-to-buffer (toa/splash-buffer))
    (toa/splash-render)
    (add-hook 'window-size-change-functions #'toa/splash-on-resize)
    (use-local-map (let ((map (make-sparse-keymap)))
                     (define-key map (kbd "q") #'toa/splash-dismiss)
                     (define-key map (kbd "RET") #'toa/splash-dismiss)
                     (define-key map (kbd "SPC") #'toa/splash-dismiss)
                     map))))

;; Show splash screen on startup in GUI mode
(when (display-graphic-p)
  (add-hook 'emacs-startup-hook #'toa/splash-show))
#+end_src
